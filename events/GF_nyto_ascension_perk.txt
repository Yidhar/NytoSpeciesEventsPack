#复制战败国科技
#This = destroyed country
#From = optional, destroyer (country)
country_event = {
	id = GFnyto_ascension_perk.130
	is_triggered_only = yes
	hide_window = yes
	immediate = {
		set_update_modifiers_batch = begin
		this = {
			save_global_event_target_as = Gfnyto_war_destroyed
		}
		from = {
			save_global_event_target_as = Gfnyto_war_murderer
		}
		every_war = {
			if = {
				limit = {
					using_war_goal = {
						type = wg_GFnyto_assimilation
						owner = event_target:Gfnyto_war_murderer
					}
				}
				event_target:Gfnyto_war_murderer = {
					copy_techs_from = {
						target = event_target:Gfnyto_war_destroyed
					}
				}
				event_target:Gfnyto_war_destroyed = {
					copy_techs_from = {
						target = event_target:Gfnyto_war_murderer
					}
				}
			}
		}
		set_update_modifiers_batch = end
	}
}

#一飞替换特质
country_event = {
	id = GFnyto_ascension_perk.777
	is_triggered_only = yes
	hide_window = yes
	immediate = {
		every_owned_pop = {
			limit = {
				is_same_species = root
				OR = {
					has_trait = trait_GF_nyto_ogas_agreement
					has_trait = trait_GF_nyto_ogas_agreement_plus
				}				
			}
			modify_species = {
				species = this
				remove_trait = trait_GF_nyto_ogas_agreement
				add_trait = trait_GF_nyto_noussphere
				effect = {
					save_event_target_as = noussphere_specoes
				}
			}
		}
		change_dominant_species = {
			species = event_target:noussphere_specoes
			change_all = yes
		}
		every_owned_leader = {
			limit = {
				is_same_species = root
				NOR = {
					has_trait = leader_admiral_trait_GF_nyto_noussphere
					has_trait = leader_general_trait_GF_nyto_noussphere
					has_trait = leader_governor_trait_GF_nyto_noussphere
					has_trait = leader_scientist_trait_GF_nyto_noussphere
					has_trait = leader_trait_GF_nyto_noussphere
				}
			}
			if = {
				limit = {
					leader_class = admiral
				}
				remove_trait = leader_admiral_trait_GF_nyto_ogas_agreement
				add_trait = leader_admiral_trait_GF_nyto_noussphere
				add_ruler_trait = leader_trait_GF_nyto_noussphere
			}
			if = {
				limit = {
					leader_class = general
				}
				remove_trait = leader_general_trait_GF_nyto_ogas_agreement
				add_trait = leader_general_trait_GF_nyto_noussphere
				add_ruler_trait = leader_trait_GF_nyto_noussphere
			}
			if = {
				limit = {
					leader_class = governor
				}
				remove_trait = leader_governor_trait_GF_nyto_ogas_agreement
				add_trait = leader_governor_trait_GF_nyto_noussphere
				add_ruler_trait = leader_trait_GF_nyto_noussphere
			}
			if = {
