#毁灭国家
country_event = {
	id = GFnytocrisis.1
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		from = {
			has_ascension_perk = ap_GF_nyto_conquer_hell
		}
		from = {
			owner = {
				has_ascension_perk = ap_GF_nyto_conquer_hell
			}
		}
	}
	immediate = {
		every_country = {
			limit = { has_event_chain = "GFnyto_crisis_chain" }
			add_event_chain_counter = {
				event_chain = "GFnyto_crisis_chain"
				counter = "GFnyto_empire_maintenance"
				amount = 500
			}
			change_variable = {
				which = "GFnyto_empire_maintenance"
				value = 500
			}
		}
	}
}

#每年人口结算,每年杀死3个涅托人口
country_event = {
	id = GFnytocrisis.12
	hide_window = yes
	trigger = {owner = {has_ascension_perk = ap_GF_nyto_conquer_hell}}
	mean_time_to_happen = {years = 1}
	immediate = {
		every_owned_planet = {
			limit = {
				num_pops > 0
			}
			while = {
				count = 3
				random_owned_pop = {
					limit = {is_exact_same_species = root}
					kill_pop = yes
				}
			}
		}
	}
}
