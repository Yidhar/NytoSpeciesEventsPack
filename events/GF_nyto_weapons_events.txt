ship_event = {
	id = gf_nyto_weapons.30002
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		exists = this
	}
	immediate = {
		switch = {
			trigger = has_ship_flag
			GFnyto_Command_ship_jump_attack ={
				repair_ship = yes
				set_disable_at_health = -1
				set_disabled = no
			}
			GF_nyto_fleet_has_shiedl_ship ={
				repair_ship = yes
				set_disabled = no
			}
		}
		if = {
			limit = {
				is_ship_size = GFnyto_stions_fortress
			}
			fire_on_action = {
				on_action = on_GF_nyto_stions_update
				scopes = {
					from = root
				}
			}
			if = {
				limit = {
					check_variable_arithmetic = {
						which = GF_nyto_is_boss_hp_mult
						subtract = 1
						value < 0
					}
				}
				set_disable_at_health = -1
				set_disabled = no
			}
			repair_ship = yes
			set_disabled = no
			subtract_variable = {
				which = GF_nyto_is_boss_hp_mult
				value = 1
			}
		}
	}
}
