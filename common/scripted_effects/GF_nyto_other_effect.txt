GF_nyto_elid_attack = {
	if = {
		limit = {
			is_scope_type = planet
		}
		if = {
			limit = {
				has_modifier = GFnyto_flower_period
			}
			change_variable = {
				which = GFnyto_Eild_level
				value = 5
			}
		}
		if = { #单位是否压缩
			limit = {
				check_variable = { #否
					which = GFnyto_Eild_level
					value < 50
				}
			}
			while = {
				count = GFnyto_Eild_level
				random_list = {
					50 ={
						create_army = {
							name = "EILD"
							owner = event_target:EILD_country
							species = event_target:ELID_species
							type = "GF_nyto_eild_army_1"
						}
						modifier ={
							divide = GFnyto_Eild_level
						}
					}
					50 = {
						create_army = {
							name = "EILD"
							owner = event_target:EILD_country
							species = event_target:ELID_species
							type = "GF_nyto_eild_army_2"
						}
						modifier ={
							subtract = GFnyto_Eild_level
						}
					}
					1 ={
						create_army = {
							name = "EILD"
							owner = event_target:EILD_country
							species = event_target:ELID_species
							type = "GF_nyto_eild_army_3"
						}						
						modifier ={
							mult = GFnyto_Eild_level
						}
					}
				}
			}

		}
		else_if = {
			limit = { 
				check_variable = { #否
					which = GFnyto_Eild_level
					value >= 50
				}
			}
			set_variable = {
				which = GFnyto_Eild_level_rar
				value = GFnyto_Eild_level
			}
			multiply_variable = {
				which = GFnyto_Eild_level_rar
				value =  0.25
			}
			if = {
				limit = { 
					check_variable = { #否
						which = GFnyto_Eild_level
						value >= 100
					}
					no_scope = {
						years_passed > 30 
					}					
				}
				event_target:EILD_country ={
					add_modifier = {
						modifier = GF_nyto_Elid_level2
						multiplier = prev.GFnyto_Eild_level_rar
					}
				}
			}
			event_target:EILD_country ={
				add_modifier = {
					modifier = GF_nyto_Elid_level
					multiplier = prev.GFnyto_Eild_level_rar
				}
			}
			while = {
				count = GFnyto_Eild_level_rar
				create_army = {
					name = "EILD"
					owner = event_target:EILD_country
					species = event_target:ELID_species
					type = "GF_nyto_eild_army_3"
				}
			}
		}
	}
}
