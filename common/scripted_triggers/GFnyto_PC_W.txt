is_nyto_country ={
	OR = {
		has_country_flag = has_nyto_technology
		has_origin = origin_GF_nyto_the_last_light
	}
}

	planet = {
		or = {
			check_variable = {
				which = GFnyto_plante_wrlv
				value < $max_wrlv$
			}
			colony_type = col_GF_nyto_capital
		}
	}
}
